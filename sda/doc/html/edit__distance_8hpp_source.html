195  int compute_matrix(int sub_cost) {
196  // Recurrence relation
197  // For each row i from i..n
198  for (int i = 1; i <= n; ++i) {
199  // For each column from j to j..m
200  for (int j = 1; j <= m; ++j) {
201  // Insertion cost
202  size_t ins = D[ i-1 ][j] + 1;
203  // Deletion cost
204  size_t del = D[i][ j-1 ] + 1;
205  // Substitution cost
206  // If A[i] equals B[j], the cost is 0
207  // If A[i] doesn't equal B[j], the cost depends on the distance metric
208  size_t sub = D[ i-1 ][ j-1 ] + (seqA[ i-1 ] == seqB[ j-1 ] ? 0 : sub_cost);
209  // Value of each cell is computed by taking the minimum of 3 possible paths
210  // through the matrix which arrive here (insertion, deletion or substition)
211  D[i][j] = min(ins,del,sub);
212  }
213  }
214  // Assert upper and lower bounds
215  assert(dist >= (n - m) || (m - n));
216  assert(dist <= ((n >= m) ? n : m));
217  if (seqA == seqB) assert(dist == 0);
218  // Termination
219  // Edit distance is found in D(n,m)
220  return dist = D[n][m];
221  }
