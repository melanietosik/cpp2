71  void compute_matrix() {
72  // Basis
73  // Initialize the first row to 0...n
74  for (int i = 0; i <= n; i++) {
75  F[i][0] = gap * i;
76  };
77  // Initialize the first column to 0...m
78  for (int j = 0; j <= m; j++) {
79  F[0][j] = gap * j;
80  };
81  // Recursion, based on the principle of optimality
82  // For each row i from i..n
83  for (int i = 1; i <= n; i++) {
84  // For each column from j to j..m
85  for (int j = 1; j <= m; j++) {
86  // Match cost
87  int match = F[i-1][j-1] + score(i,j);
88  // Deletion cost
89  int del = F[i-1][j] + gap;
90  // Insertion cost
91  int ins = F[i][j-1] + gap;
92  // For each cell, F(i,j) contains maximum score among all paths
93  F[i][j] = max(match,del,ins);
94  }
95  }
96  }

109  void trace_back() {
110  int i = n;
111  int j = m;
112  // Maximum alignment score
113  max_score = F[i][j];
114  // Start at the end of the input sequences
115  while (i > 0 || j > 0) {
116  // If match, A[i] and B[j] are aligned
117  if ( i>0 && j>0 && F[i][j] == ( F[i-1][j-1] + score(i,j))) {
118  alignA += seqA[i-1];
119  alignB += seqB[j-1];
120  i--;
121  j--;
122  }
123  // If delete, A[i] is aligned with a gap
124  else if ( i>0 && F[i][j] == F[i-1][j] + gap) {
125  alignA += seqA[i-1];
126  alignB += "-";
127  i--;
128  }
